100x = 24.000

x = 24.000/100

x = 240ml

[tex] = \frac{200ml \: susu}{150ml \: kopi} [/tex]

[tex] = \frac{200}{150} [/tex]

[tex] = \frac{200 \div 50}{150 \div 50} [/tex]

[tex] = \frac{40}{30} [/tex]

[tex] \: [/tex]

[tex]sehingga[/tex]

[tex]susu \: yang \: dibutuhkan[/tex]

[tex] = kopi \times perbandingan \: komposisi[/tex]
